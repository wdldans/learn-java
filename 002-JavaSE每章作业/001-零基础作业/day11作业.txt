public class Test{
	public static void main(String[] args){
		int a = 100;
		int b = 200;
		int retValue = m1(a, b);
		System.out.println(retValue);
	}
	public static int m1(int x, int y){
		int k = x * 10;
		int m = y * 10;
		int result = m2(k, m);
		return result;
	}
	public static int m2(int c, int d){
		int e = c / 2;
		int f = d / 2;
		int i = e + f;
		return i;
	}
}
